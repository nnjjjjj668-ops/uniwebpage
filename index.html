1 — Architecture & Confirmed Tech Stack

Item 1 of 4 
FRONTEND
  ├── Vanilla HTML5 / CSS3 / ES6 JS         # Assignment mandated. No framework.
  ├── Groq JS SDK (fetch-based)              # In-browser API calls to Groq
  └── Google Fonts CDN                       # Typography only — zero JS weight

BACKEND (Production Recommendation)
  ├── Runtime   : Node.js 20 LTS
  ├── Framework : Next.js 14 (App Router)    # API Routes for Groq proxy, auth
  ├── ORM       : Prisma 5                   # Schema-first, type-safe DB access
  ├── Database  : PostgreSQL 16              # Reliable for relational data
  └── Auth      : NextAuth.js / JWT          # Session management

AI INTEGRATION
  ├── Provider  : Groq Cloud                 # Ultra-fast inference (LPU hardware)
  ├── Model     : llama3-8b-8192             # Best speed/quality for MCQ tasks
  └── Fallback  : mixtral-8x7b-32768        # Longer context if needed

DEPLOYMENT
  ├── Frontend  : Vercel                     # Zero-config Next.js hosting
  └── Database  : Supabase / Neon (PG)       # Managed PostgreSQL

model User {
  id            String          @id      @default(cuid())
  enrollmentNo  String          @unique
  email         String          @unique
  passwordHash  String
  firstName     String
  lastName      String
  gender        Gender
  role          Role             @default(STUDENT)
  programme     String
  semester      Int
  cgpa          Float?
  skills        String[]
  createdAt     DateTime         @default(now())
  updatedAt     DateTime         @updatedAt

  quizResults   QuizResult[]
  resumeProfile ResumeProfile?
  events        CalendarEvent[]
}

enum Gender { MALE FEMALE OTHER PREFER_NOT_DISCLOSE }
enum Role   { STUDENT FACULTY ADMIN }

// ── COURSE MATERIAL ─────────────────────────────────────
model CourseMaterial {
  id          String       @id     @default(cuid())
  subjectCode String
  subjectName String
  semester    Int
  department  String
  module      String
  resourceType ResourceType
  title       String
  url         String
  fileSize    Int?         // bytes
  uploadedAt  DateTime     @default(now())
  isActive    Boolean      @default(true)

  @@index([subjectCode])
  @@index([semester, department])
}

enum ResourceType { PDF VIDEO AUDIO LINK NOTES }

// ── QUIZ RESULT ─────────────────────────────────────────
model QuizResult {
  id          String   @id     @default(cuid())
  userId      String
  topic       String
  score       Int
  totalQ      Int
  answers     Json     // { qIndex: chosenOption }[]
  generatedAt DateTime @default(now())
  timeTaken   Int?     // seconds

  user        User     @relation(fields: [userId], references: [id])

  @@index([userId])
}

// ── CALENDAR EVENT ──────────────────────────────────────
model CalendarEvent {
  id          String    @id     @default(cuid())
  userId      String
  title       String
  description String?
  eventDate   DateTime
  location    String?
  eventType   EventType @default(GENERAL)
  isAllDay    Boolean   @default(false)
  createdAt   DateTime  @default(now())

  user        User      @relation(fields: [userId], references: [id])

  @@index([userId, eventDate])
}

enum EventType { EXAM ASSIGNMENT LECTURE GENERAL }

// ── RESUME PROFILE ───────────────────────────────────────
model ResumeProfile {
  id           String   @id     @default(cuid())
  userId       String   @unique
  targetRole   String   @default("Data Analyst")
  summary      String?
  skills       String[] // e.g. ["Python","SQL","ML"]
  educationJson Json    // structured degree entries
  projectsJson  Json    // structured project entries
  certsJson     Json?   // certifications
  updatedAt     DateTime @updatedAt

  user         User     @relation(fields: [userId], references: [id])
}

4 — Next.js API Route: Groq Quiz Generator

Item 4 of 4 — app/api/quiz/route.ts 
// app/api/quiz/route.ts
// Next.js 14 App Router · Groq LLaMA 3 · 5-question MCQ generator

import { NextRequest, NextResponse } from 'next/server';
import Groq from 'groq-sdk';

const groq = new Groq({ apiKey: process.env.GROQ_API_KEY });

/** Shape of a single MCQ question returned to the client */
export interface QuizQuestion {
  question: string;
  options:  [string, string, string, string];
  answer:   number;  // 0-indexed correct option
}

/** POST /api/quiz — Body: { topic: string } */
export async function POST(req: NextRequest) {
  try {
    const { topic }: { topic: string } = await req.json();

    if (!topic?.trim()) {
      return NextResponse.json(
        { error: 'topic is required' },
        { status: 400 }
      );
    }

    const prompt = `
You are an academic quiz generator for a college-level Computer Science course.
Generate exactly 5 multiple-choice questions about: "${topic}".

Rules:
- Each question must have exactly 4 answer options (A, B, C, D).
- Exactly one option must be correct.
- Questions should be clear, unambiguous, and test genuine understanding.
- Vary difficulty: 2 easy, 2 medium, 1 challenging.

Respond ONLY with a valid JSON array — no markdown, no preamble.
Format:
[
  {
    "question": "Question text here?",
    "options": ["Option A", "Option B", "Option C", "Option D"],
    "answer": 0
  }
]
The "answer" field is the 0-based index of the correct option.
    `.trim();

    const completion = await groq.chat.completions.create({
      model:       'llama3-8b-8192',
      temperature: 0.4,
      max_tokens:  1200,
      messages: [
        {
          role:    'system',
          content: 'You are a precise JSON generator. Output only valid JSON.',
        },
        {
          role:    'user',
          content: prompt,
        },
      ],
    });

    const raw = completion.choices[0]?.message?.content ?? '';

    // Strip markdown fences if the model wraps output despite instructions
    const clean = raw.replace(/```(?:json)?|```/g, '').trim();

    const questions: QuizQuestion[] = JSON.parse(clean);

    if (!Array.isArray(questions) || questions.length === 0) {
      throw new Error('Malformed response from model');
    }

    return NextResponse.json({ topic, questions }, { status: 200 });

  } catch (err: unknown) {
    console.error('[quiz/route]', err);
    return NextResponse.json(
      { error: 'Failed to generate quiz. Please try again.' },
      { status: 500 }
    );
  }
}
